import 'dart:async';
import 'package:flutter/material.dart';
import 'package:flutter_map/flutter_map.dart';
import 'package:flutter_map_cancellable_tile_provider/flutter_map_cancellable_tile_provider.dart';
import 'package:flutter_map_location_marker/flutter_map_location_marker.dart';
import 'package:flutter_map_marker_cluster/flutter_map_marker_cluster.dart';
import 'package:hooks_riverpod/hooks_riverpod.dart';
import 'package:latlong2/latlong.dart';
import 'package:travelapp/src/features/map/map.dart';
import 'package:travelapp/src/features/touristic_sites/touristic_sites.dart';
import 'package:url_launcher/url_launcher.dart';

class MapWidget extends HookConsumerWidget {
  const MapWidget({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final PopupController popupController = ref.watch(popupControllerProvider);
    final MapController mapController = MapController();

    // Debounce timer for updating the sites
    Timer? debounceTimer;

    // Function to update the sites based on the current visible bounds
    Future<void> updateSites() async {
      // Hide all popups before updating
      popupController.hideAllPopups();

      // Get the current visible bounds
      final LatLngBounds visibleBounds = mapController.camera.visibleBounds;

      // Fetch the sites within the current visible bounds
      final TouristicSitesRepository touristicSitesRepository =
          ref.read(touristicSitesRepositoryProvider);
      final List<TouristicSiteSimple> sites = await touristicSitesRepository
          .getTouristicSitesSimpleOnMap(visibleBounds);

      // Update the provider with the fetched sites
      ref
          .read(touristicSiteSimpleListProvider.notifier)
          .touristicSiteListSimple = sites;
    }

    return FlutterMap(
      mapController: mapController,
      options: MapOptions(
        initialCenter: const LatLng(51.509364, -0.128928),
        initialZoom: 3.2,
        onPositionChanged: (MapCamera position, bool hasGesture) {
          // Set the zoom level in the provider
          ref.read(mapZoomProvider.notifier).zoom = position.zoom;

          // If there is a gesture, start or reset the debounce timer
          if (hasGesture) {
            // Cancel the existing timer if it’s still running
            debounceTimer?.cancel();

            // Set a new debounce timer to update the sites
            debounceTimer = Timer(const Duration(milliseconds: 300), () {
              updateSites();
            });
          }
        },
        onTap: (_, __) {
          // Close popups on map tap
          popupController.hideAllPopups();
        },
        onMapReady: () {
          ref.read(mapControllerProvider.notifier).state = mapController;
        },
      ),
      children: <Widget>[
        TileLayer(
          tileProvider: CancellableNetworkTileProvider(),
          urlTemplate: 'https://tile.openstreetmap.org/{z}/{x}/{y}.png',
          userAgentPackageName: 'com.example.app',
        ),
        RichAttributionWidget(
          showFlutterMapAttribution: false,
          attributions: <SourceAttribution>[
            TextSourceAttribution(
              'OpenStreetMap contributors',
              onTap: () =>
                  launchUrl(Uri.parse('https://openstreetmap.org/copyright')),
            ),
          ],
        ),
        const SightMarkers(),
        const RouteSightsMarker(),
        const RoutePlaceMarker(),
        const RouteSightConnections(),
        const RoutePlaceConnections(),
        CurrentLocationLayer(
          alignPositionOnUpdate: AlignOnUpdate.never,
          alignDirectionOnUpdate: AlignOnUpdate.never,
          style: const LocationMarkerStyle(
            marker: DefaultLocationMarker(
              child: Icon(
                Icons.navigation,
                color: Colors.white,
              ),
            ),
            markerSize: Size(40, 40),
            markerDirection: MarkerDirection.heading,
          ),
        )
      ],
    );
  }
}

import 'package:flutter/material.dart';
import 'package:flutter_map/flutter_map.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:latlong2/latlong.dart';
import 'package:travelapp/src/features/routes/routes.dart' as r;

Map<String, dynamic> extractMapElements(List<r.Route> routeList) {
  final List<Marker> placeMarkers = <Marker>[];
  final List<Marker> sightMarkers = <Marker>[];
  final List<Polyline<Object>> sightConnections = <Polyline<Object>>[];
  final List<Polyline<Object>> placeConnections = <Polyline<Object>>[];

  for (final r.Route route in routeList) {
    route.itinerary.asMap().forEach((int index, dynamic item) {
      if (item['type'] == 'place') {
        final LatLng placeLatLng =
            LatLng(item['latitude'] as double, item['longitude'] as double);
        placeMarkers.add(
          Marker(
            width: 80.0,
            height: 80.0,
            point: placeLatLng,
            child: const Icon(Icons.location_on, color: Colors.blue, size: 40),
          ),
        );

        for (final dynamic site in item['sites_to_visit'] as List<dynamic>) {
          final LatLng sightLatLng =
              LatLng(site['latitude'] as double, site['longitude'] as double);
          sightMarkers.add(
            Marker(
              width: 80.0,
              height: 80.0,
              point: sightLatLng,
              child:
                  const Icon(Icons.location_on, color: Colors.black, size: 40),
            ),
          );

          sightConnections.add(
            Polyline<Object>(
              points: <LatLng>[placeLatLng, sightLatLng],
              strokeWidth: 3.0,
              color: Colors.black,
            ),
          );
        }
      }
      if (item['type'] == 'transfer') {
        final int idxStartPlace = index - 1;
        final int idxStopPlace = index + 1;

            route.itinerary[idxStartPlace]['longitude'] as double);
        final LatLng endLatLng = LatLng(
            route.itinerary[idxStopPlace]['latitude'] as double,
            route.itinerary[idxStopPlace]['longitude'] as double);
        placeConnections.add(
          Polyline<Object>(
            points: <LatLng>[startLatLng, endLatLng],
            strokeWidth: 5.0,
            color: Colors.blue,
          ),
        );
      }
    });
  }

  return <String, Widget>{
    'placeMarkers': MarkerLayer(markers: placeMarkers),
    'sightMarkers': MarkerLayer(markers: sightMarkers),
    'sightConnections': PolylineLayer<Object>(polylines: sightConnections),
    'placeConnections': PolylineLayer<Object>(polylines: placeConnections),

}

@riverpod
RoutesRepository routesRepository(RoutesRepositoryRef _) => RoutesRepository();

class RoutesRepository {
  final SupabaseClient _client = Supabase.instance.client;
  Future<List<Route>> getRoutesOnMap(
      LatLngBounds visibleBounds, int minDays, int maxDays) async {
    final List<Map<String, dynamic>> data =
        await _client.rpc('get_routes_on_map', params: <String, dynamic>{
      'max_lat': visibleBounds.north,
      'max_long': visibleBounds.east,
      'min_lat': visibleBounds.south,
      'min_long': visibleBounds.west,
      'min_days': minDays,
      'max_days': maxDays
    });

    final List<Route> routes = data.map((Map<String, dynamic> item) {
      return Route(
        id: item['id'] as int,
        name: item['name'] as String,
        description: item['description'] as String,
        days: item['days'] as int,
        country: item['country'] as String,
        tags: item['tags'] as List<dynamic>,
        itinerary: item['itinerary'] as List<dynamic>,
      );
    }).toList();
    return routes;
  }
}
